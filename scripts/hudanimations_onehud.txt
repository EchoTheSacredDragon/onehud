//Health Bonus Pulse
event HudHealthBonusPulse
{
	Animate	PlayerStatusHealthBonusImage	Alpha	"0"		Linear 0.0 0.0

	Animate PlayerStatusHealthValueGlow FgColor "HP Buff" Linear 0.0 0.2
	Animate PlayerStatusHealthValueGlow FgColor "HP Buff 2" Linear 0.2 0.4

	Animate PlayerStatusHealthValue 	FgColor "HP Buff" Linear 0.0 0.2
	Animate PlayerStatusHealthValue 	FgColor "HP Buff 2" Linear 0.2 0.4

	Animate PlayerStatusHealthValueShadow 	FgColor "Blank" Linear 0.0 0.2

	RunEvent HudHealthBonusPulseLoop 0.4
}

// call to loop HudHealthBonusPulse
event HudHealthBonusPulseLoop
{
	RunEvent HudHealthBonusPulse 0.0
}

event HudHealthBonusPulseStop
{
	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0


	Animate PlayerStatusHealthValueGlow FgColor "Blank" Accel 0.0 0.2
	Animate PlayerStatusHealthValue FgColor "HP" Accel 0.0 0.2
	Animate PlayerStatusHealthValueShadow 	FgColor "CBlackShadow" Linear 0.0 0.2
}

//Low Ammo Pulse
event HudLowAmmoPulse
{
	Animate AmmoInClip FgColor "LowAmmo1" Linear 0.0 0.2
	Animate AmmoInClip FgColor "LowAmmo2" Linear 0.2 0.4
	Animate AmmoInClipShadow FgColor "Blank" Linear 0.0 0.2

	Animate AmmoInClipGlow FgColor "LowAmmo1" Linear 0.0 0.2
	Animate AmmoInClipGlow FgColor "LowAmmo2" Linear 0.2 0.4

	Animate AmmoNoClip FgColor "LowAmmo1" Linear 0.0 0.2
	Animate AmmoNoClip FgColor "LowAmmo2" Linear 0.2 0.4
	Animate AmmoNoClipShadow FgColor "Blank" Linear 0.0 0.2

	RunEvent HudLowAmmoPulseLoop 0.4
}

// call to loop HudLowAmmoPulse
event HudLowAmmoPulseLoop
{
	RunEvent HudLowAmmoPulse 0.0
}

event HudLowAmmoPulseStop
{
	StopEvent HudLowAmmoPulse 0.0
	StopEvent HudLowAmmoPulseLoop 0.0

	Animate AmmoInClip FgColor "Ammo In Clip" Linear 0.0 0.2
	Animate AmmoInClipShadow FgColor "CBlackShadow" Linear 0.0 0.2
	Animate AmmoInClipGlow FgColor "Blank" Linear 0.0 0.2

	Animate AmmoNoClip FgColor "Ammo In Clip" Linear 0.0 0.2
	Animate AmmoNoClipShadow FgColor "CBlackShadow" Linear 0.0 0.2
}
